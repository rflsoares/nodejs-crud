# Instalando o ambiente
### Mysql
$ sudo pacman -S mariadb
$ sudo mysql_install_db --user=mysql --basedir=/usr --datadir=/var/lib/mysql
$ sudo systemctl enable mysqld
$ sudo systemctl start mysqld
$ sudo mysql_secure_installation
